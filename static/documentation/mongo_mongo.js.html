/**
 * Check itemref collision.
 * @param {object} itemref - example input: {itemref: '098p0p00'}
 * @description Checks the database if a given itemref exist.
 * @return {Promise<Boolean>} Resolves a Boolean indicating whether to generate a new
 * itemref or not.
 */
  exports.checkCollision =  (itemref)=>{
    let promise = new Promise((resolve, reject)=>{

      stock.findOne(itemref, function(err, results){
        if (err) reject(err);
        if(!results){
          resolve(false)
        }
        else{
          resolve(true);
        }
      });
    });
    return promise;
  }

  function exists(field, name){
    let query = { [field]: name};
    
    let promise = new Promise((resolve, reject)=>{

        if (err) reject(err);
        if(!results){
          resolve(false);

          resolve(true)

  /**
   * Add new and save OR update
   * @param {object} { dimensions: [ 290, 40, 290 ],weight: 900, name: 'large plate'}
   * @description Checks the database if a given item type exist, if it does
   * it updates it, else it creates a new item
   * @return {object} Mongo result modified true or not
   */
  exports.updateItemType = (req, callback) => {
    
    
    let promise = new Promise((resolve, reject)=>{
      (async function saveUpdate(){
        let type = req.updates.itemtype;
        let name = req.updates.name;
        delete req.updates.itemtype;

        let exist = await exists([type]+'.name', req.updates.name);
        let getFiles = await getArrayWithAllPathsInDir(type, name);


        function getArrayWithAllPathsInDir(type, name){
          let promise = new Promise((resolve, reject)=>{
            if(type === 'item type'){
              resolve();
            }
            let path = './static/images/stock' + '/' + type + '/' + name + '/';
            fs.readdir(path, function (err, files) {
              if (err) reject(err)
              else{
                
                let extPath = path.substr(8, path.length);
                let fullPaths = [];
                files.forEach((item, i) => {
                  fullPaths.push(extPath.concat(item));
                  if(i === files.length-1){
                    
                    req.updates['image links'] = fullPaths;
                    resolve();
                  }
                });

              }
            });
          });
          return promise;
        }


        if(!exist){
          stock.updateOne(
            { 'master item type': true},
              {$push:
                { [type]:
                  req.updates }},
                  {upsert: true},
            function(err, result){
              if (err) reject(err);
              resolve(result);
            });
        }
        else{
          let query = { 'master item type': true,[type+'.name']: name};
          let update = {[type+'.$']: req.updates};

          stock.findOneAndUpdate(
            query,
            {$set:
              update
            }
            ,
            {upsert: true}, function(err, result){
              if (err) reject(err);
              resolve(result);
            });
        }
      })();
    });
    return promise;
  }

  exports.updateTheme = (req, callback) => {
    
    let promise = new Promise((resolve, reject)=>{
      (async function saveUpdate(){
        let exist = await exists('theme.name', req.theme.value);
        if(!exist){
          stock.updateOne(
            { 'master item type': true},
              {$push:
                { 'theme':
                  {name: req.theme.value, images: req.theme['image links'] || []} }},
                  {upsert: true},
            function(err, result){
              if (err) reject(err);
              resolve(result);
            });
        }
        else{
          stock.updateOne(
            { 'master item type': true,
            'theme':
              //{ $elemMatch:
                { 'theme.name': req.theme.value }
              //}
            },
            {$set:
              {'theme.$' : req.theme}
            },{upsert: true}, function(err, result){
              if (err) reject(err);
              resolve(result);
            });
        }
      })();
    });
    return promise;
  }
